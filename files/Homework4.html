predictclass= function(predicted, numberofclasses)
{
  predictedclass=c()
  for(j in 1:length(predicted[,1,]))
  {
    predictedclass[j]=which.max(predicted[j,,])[[1]] #gives the class of the highest probability
  }
  return(predictedclass)
}  
decision_trees <- function(train.data) #returns best complexity parameter and the minimal number of observations per tree leaf 
{
  treetune=tune.rpart(class~.,data=train.data,cp=c(0.1,0.05,0.01,0.001,0.005,0.0001),minsplit = c(5,10,15,20,25,30))
  bestparameters=c()
  bestparameters[1]=treetune$best.parameters[[1]]
  bestparameters[2]=treetune$best.parameters[[2]]
  return(bestparameters)
}
support_vector_machine <- function(train.data) # returns best kernel type, its parameter and cost value
{
  svmPolytune=tune.svm(class~.,data=train.data, kernel="polynomial",cost=c(0.01,1,1000),degree=c(5,8,10), scale=FALSE)
  svmGaustune=tune.svm(class~.,data=train.data, kernel="radial",cost=c(0.01,1,1000),gamma=c(0.25,0.5,1,2), scale=FALSE)
  svmPolydegree=svmPolytune$best.parameters[[1]]
  svmPolycost=svmPolytune$best.parameters[[2]]
  svmGausgamma=svmGaustune$best.parameters[[1]]
  svmGauscost=svmGaustune$best.parameters[[2]]
  if(svmPolytune$best.performance<=svmGaustune$best.performance)
  {
    return(parameters=c(1,svmPolydegree,svmPolycost))
  }
  if(svmPolytune$best.performance>svmGaustune$best.performance)
  {
    return(parameters=c(2,svmGausgamma,svmGauscost))
  }
}
randomforest <- function(train.data,train.class,train.datanoclass) # returns the number of random sample of m features
{
  forest.tune<-rfcv(train.datanoclass,train.class,cv.fold = 10, mtry=function(p) max(1, floor(sqrt(p))),recursive=FALSE,ntree=500,nodesize=5)
  m=which.min(forest.tune$error.cv)[1]
  mbest=as.numeric(names(m)[1])
  return(mbest)
}

penalized_regression=function(train.data,binary)#returns best lamda value based on the cross validation
{
  train.class=train.data$class
  glm_train_data=train.data[,-ncol(train.data)]
  alpha=1
  nlambda=50
  set.seed(1)
  nFolds=10
  nofReplications=2
  cvindices=generateCVRuns(train.class,nofReplications,nFolds,stratified=TRUE)
  glmnet_alldata = glmnet(as.matrix(glm_train_data),  as.factor(train.class), family="multinomial", alpha = alpha, nlambda=nlambda)
  lambda_sequence = glmnet_alldata$lambda
  bestaccuracy=0
  bestlambda=0
  for(i in 1:nofReplications) {
    thisReplication=cvindices[[i]]
    for(j in 1:nFolds){
      testindices=order(thisReplication[[j]])
      cvtrain=glm_train_data[-testindices,]    
      cvtrainclass=train.class[-testindices]   
      cvtest=glm_train_data[testindices,]
      cvtestclass=train.class[testindices] 
      inner_cv_glmnet_fit = glmnet(as.matrix(cvtrain),as.factor(cvtrainclass),family="multinomial", alpha = alpha,lambda=lambda_sequence)
      valid_pred = predict(inner_cv_glmnet_fit, as.matrix(cvtest), s = lambda_sequence, type = "response")
      order_of_class=attr(valid_pred,'dimnames')[[2]]
      foldresult=rbindlist(lapply(c(1:length(lambda_sequence)),function(x) { data.table(repl=i,fold=j,lambda=lambda_sequence[x],valid_pred[,order_of_class,x],result=cvtestclass)}))
      for(i in 1:length(foldresult$lambda))
      {
        foldresult[i,testresult:=which.max(foldresult[i, 4:(ncol(foldresult)-2)])[[1]]]
      }
      if(binary==1)
      {
        foldresult$testresult=foldresult$testresult-1
      }
      accuracy=c()
      begin=1
      for(k in 1:length(lambda_sequence))
      {
        accuracy[k]=confusionMatrix(foldresult$testresult[begin:(k*dim(cvtest)[1])],foldresult$result[begin:(k*dim(cvtest)[1])])$overall[[1]]
        begin=k*dim(cvtest)[1]+1
      }
      if(max(accuracy)>=bestaccuracy)
      {
        bestaccuracy=max(accuracy)
        bestlambda=foldresult$lambda[which.max(accuracy)*length(lambda_sequence)]
      }
    }
  }
  return(bestlambda)
}
crossValidationTrain=function(train.data=project.train, parameters,classifier) #returns mean train accuracy based on cross validation
{
  set.seed(1)
  nFolds=10
  nofReplications=2
  cvindices=generateCVRuns(train.data$class,nofReplications,nFolds,stratified=TRUE)
  accuracy=c()
  glmtrain.class=train.data$class
  alpha=1
  glmtrain.data=train.data[,-ncol(train.data)]
  k=1
  for(i in 1:nofReplications) {
    thisReplication=cvindices[[i]]
    for(j in 1:nFolds){
      if(classifier==1) #penalized regression
      {
        testindices=order(thisReplication[[j]])
        cvtrain=glmtrain.data[-testindices,]    
        cvtrainclass=glmtrain.class[-testindices]   
        cvtest=glmtrain.data[testindices,]
        cvtestclass=glmtrain.class[testindices] 
        glmnet_fit = glmnet(as.matrix(cvtrain),as.factor(cvtrainclass),family="multinomial", alpha = alpha,lambda=parameters[1])
        valid_pred = predict(glmnet_fit, as.matrix(cvtest), type = "response")
        valid_pred=predictclass(valid_pred,length(unique(cvtrainclass)))
        if(parameters[2]==1)
        {
          valid_pred=valid_pred-1
        }
      }
      if(classifier==2) #decision trees
      {
        testindices=order(thisReplication[[j]])
        cvtrain=train.data[-testindices,]    
        cvtest=train.data[testindices,]
        tree=rpart(class~.,data=cvtrain,method="class",control=rpart.control(xval=10,cp=parameters[2],minbucket = parameters[1]))
        valid_pred=predict(tree,newdata=cvtest[,-(ncol(cvtest))],type="class")
        cvtestclass=cvtest[,(ncol(cvtest))]
      }
      if(classifier==3) #support vector machines
      {
        testindices=order(thisReplication[[j]])
        cvtrain=train.data[-testindices,]    
        cvtest=train.data[testindices,]
        if(parameters[1]==2)
        {
          SVM=svm(class~.,type="C-classification", data=cvtrain , kernel ="radial", gamma=parameters[2], cost =parameters[3], scale=FALSE)
        }
        if(parameters[1]==1)
        {
          SVM=svm(class~.,type="C-classification", data=cvtrain , kernel ="polynomial", degree=parameters[2], cost =parameters[3], scale=FALSE)
        }
        valid_pred=predict(SVM,cvtest[,-(ncol(cvtest))])
        cvtestclass=cvtest[,(ncol(cvtest))]
      }
      if(classifier==4) #random forest
      {
        testindices=order(thisReplication[[j]])
        cvtrain=train.data[-testindices,]    
        cvtest=train.data[testindices,]
        RF<-randomForest(class~.,data=cvtrain,mtry=parameters,importance=TRUE,ntree=500,nodesize=5)
        valid_pred<-predict(RF,newdata=cvtest[,-(ncol(cvtest))])
        cvtestclass=cvtest[,(ncol(cvtest))]
      }
      if(classifier==5) #stochastic gradient boosting
      {
        testindices=order(thisReplication[[j]])
        cvtrain=train.data[-testindices,]    
        cvtest=train.data[testindices,]
        if(parameters[1]==1)
        {
          GBM=gbm(class~., data=cvtrain,distribution = "bernoulli", n.trees = parameters[4],
                  interaction.depth = parameters[2], n.minobsinnode = 10, shrinkage =parameters[3], cv.folds = 0)
          valid_pred=predict(GBM,cvtest[,-(ncol(cvtest))],type = "response",parameters[4])
          valid_pred[valid_pred>0.5]=1
          valid_pred[valid_pred<=0.5]=0
        }
        if(parameters[1]==2)
        {
          GBM=gbm(class~., data=cvtrain, distribution = "multinomial", n.trees = parameters[4],
                  interaction.depth = parameters[2], n.minobsinnode = 10, shrinkage =parameters[3], cv.folds = 0)
          valid_pred=predict(GBM,cvtest[,-(ncol(cvtest))],type = "response",parameters[4])
          valid_pred=predictclass(valid_pred,length(unique(cvtest$class)))
        }
        cvtestclass=cvtest[,(ncol(cvtest))]
      }
      accuracy[k]=confusionMatrix(valid_pred,cvtestclass)$overall[[1]]
      k=k+1
    }
  }
  return(mean(accuracy))
}
